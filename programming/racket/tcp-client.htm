TCP client

(require racket/tcp)

(define-values (in out) (tcp-connect  "192.168.0.15" 3333))
(define (msg line)
  (displayln line out)
  (flush-output out)
  (display "Response: ")
  (displayln (read-line in)))
(msg "hello world")
(msg "ESP32 rocks")
(close-input-port in)
(close-output-port out)
