Processes

Call the ls command and print the directory entries: 
#lang racket

(require data/queue)
(define pr (process* "/bin/ls" "-hal" "/"))
(define iport (first pr))
(define q (make-queue))
(let loop ()
  (define node (read-line iport))
  (unless (eof-object? node)
    (enqueue! q node)
    (loop)))

(for ((node q))
  (display node)
  (newline))
(close-input-port iport)
