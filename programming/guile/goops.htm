goops

Example

(use-modules (oop goops))

(define-class <cls> (<object>)
  x)

(define c1 (make <cls>))
(slot-set! c1 'x 0)
(display (slot-ref c1 'x)) ; outputs 0

(define (adj c)
  (slot-set! c 'x 1))

(adj c1)
(display (slot-ref c1 'x)) ; outputs 1
