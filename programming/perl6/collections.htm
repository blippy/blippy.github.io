Hashes

my %h = %( a => 1, v => 2);
say %h<b>; # OUTPUT: 2
%h<b> = 4;
say %h<b>; # OUTPUT: 4


my %h = %(12=>13, 14=>15);
my $f = 12;
%h{$f}; # 13
%h<16>; # (Any)
%h<16> || 666; # 666
%h<14> || 666; # 15
