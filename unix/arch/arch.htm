Arch Linux

Preliminary installation instructions are available by typing 
  wget http://www.markcarter.me.uk/unix/arch.txt

keyboard

locale # display current locale

nano /etc/locale.gen # uncomment all en_GB* for Britain
locale-gen

nano /etc/locale.conf
LANG=en_GB.UTF-8

# set console keymap:
nano /etc/vconsole.conf
KEYMAP=uk

# set X11 keymap:
localectl set-x11-keymap gb

network

  ip addr show # show your IP address
  ip link set enp3s0 down # take down ethernet interface

timezone for UK

  ln -s /usr/share/zoneinfo/Europe/London /etc/localtime
