Debian 10 server

Edit ~/.vnc/xstartup: 
#!/bin/env bash
exec /usr/bin/mate-session &
Start vnc: 
	vncserver :1  -geometry 1920x1080 -localhost no
or
	vncserver  -geometry 1920x1080 -localhost no
or
	vnc -c # my own script
