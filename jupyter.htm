# Install into a virtual environment
virtualenv --version # check existence
pip3 install --user virtualenv
mkdir -p ~/tools
cd ~/tools
virtualenv --no-site-packages jupyter
cd jupyter/bin
source activate
pip3 install jupyter

# install micropython kernel
cd ~/tools/jupyter
git clone https://github.com/goatchurchprime/jupyter_micropython_kernel.git
pip3 install -e jupyter_micropython_kernel
python3 -m jupyter_micropython_kernel.install
jupyter kernelspec list

# Run it from a different shell
jup # my own script
